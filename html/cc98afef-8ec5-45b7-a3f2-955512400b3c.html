\( \begin{cases} \begin {align}4x + 3y &= 10\\ -4x + 5y &= 6 \end{align} \end{cases} \)

\( \begin {align}4x + 3y &= 10\\ -4x + 5y &= \hspace{2mm}6 \quad+\\ \overline {\quad 0 + 8y} &\overline{ \hspace{1mm}= 16 \qquad}\\ y &= 2 \end{align} \)

\( \begin {align}4x + 3(2) &= 10\\ 4x + 6 &= 10\\ 4x &= 4\\ x &= 1 \end{align} \)
