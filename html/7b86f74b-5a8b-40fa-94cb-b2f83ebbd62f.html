1. Solve

\( \left\{ \begin{array}{c l} –2x + 5y = –42\\ 7x + 2y = 30 \end{array}\right. \)

  • (0,15)
  • (11,−4)
  • (1,−8)
  • (6,−6)

\(- 2( –2x+5y=–42)=4x-10y=84 \)

\(5(7x+2y=30)=35x+10y=150\)

\(\begin{array}{r}4x-10y&=&84 \\35x+10y&=&150 \\ \hline 39x&=&234 \\x&=&6 \end{array}\)

\(\begin{array}{rcl}7x+8y&=&4 \\(7(6)+2y&=&30 \\42+2y&=&30 \\2y&=&-12 \\y&=&-6 \\\end{array}\)
(6,-6)
